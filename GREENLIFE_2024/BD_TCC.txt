CREATE DATABASE GL_BD;

USE GL_BD;
GO

-- Criar a tabela Cadastro
CREATE TABLE [dbo].[Cadastro] (
    [Id]    INT            IDENTITY (1, 1) NOT NULL,
    [Nome]  VARCHAR (100) NOT NULL,
    [Email] VARCHAR (100) UNIQUE NOT NULL,
    [Senha] VARCHAR (100) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
GO
--Criar a tabela Admin
CREATE TABLE [dbo].[Admin] (
    [Id]    INT            IDENTITY (1, 1) NOT NULL,
    [Nome]  NVARCHAR (100) NOT NULL,
    [Email] NVARCHAR (100) UNIQUE NOT NULL,
    [Senha] NVARCHAR (100) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

GreenLife :
CREATE TABLE [dbo].[Plantio] (
    [Id]           INT           IDENTITY (1, 1) NOT NULL,
    [NomePlantio]  VARCHAR (100) NOT NULL,
    [DataPlantio]  DATE      NOT NULL,
    [Tipo]         VARCHAR (100) NOT NULL,
    [Semente]      VARCHAR (100) NOT NULL,
    [Local]        VARCHAR (100) NOT NULL,
    [DataColheita] DATE      NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
